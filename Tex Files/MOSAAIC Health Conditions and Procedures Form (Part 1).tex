\subsection*{\FetchText{HCP0.}}
\label{subsec:HCP1}
\FetchText{HCP0.1}%




\vlist{1.} \item Since the date of your in-person MOSAAIC exam, has a health care provider said that you had any of the following health conditions? \textit{(Select all that apply)}
    \begin{enumerate}[left = 0 em, align = parleft]
    \item Heart attack (also called ``myocardial infarction'') \ra \textbf{Complete Myocardial Infarction and CAD Form on page~\pageref{Heart attack or myocardial infarction}}
    \item Chest pain, angina, or coronary artery disease \ra \textbf{Complete Myocardial Infarction and CAD Form on page~\pageref{Heart attack or myocardial infarction}}
    \item Heart Failure \ra \textbf{Complete Heart Failure Form on page~\pageref{Heart failure}}
    % \par\begin{flushright}\begin{tcolorbox}[colback = white, width = .7\textwidth]\hspace{-.3cm}\enode%
    % If yes, on what date did you receive your diagnosis? \newline\hxdate\end{tcolorbox}\end{flushright}
    % %~\ra Date of Diagnosis:\vspace{.5\baselineskip}
    % % \par \begin{flushright}\MMDDYYYY (MM/DD/YYYY)\end{flushright}
    \item Atrial fibrillation or atrial flutter \ra \textbf{Complete Atrial Fibrillation Form on page~\pageref{Atrial fibrillation}}
    % \itemnode Atrial flutter \diagnosisdate
    %\item Other heart disease %\diagnosisdate
    \item Stroke \ra \textbf{Complete Stroke Form on page~\pageref{Stroke}}
    \item Mini-stroke or transient ischemic attack \ra \textbf{Complete TIA Form on page~\pageref{Transient Ischemic Attack}} %\diagnosisdate
    \item Blood clot in your leg or lung requiring blood thinning medicine~\ra~\textbf{Complete PE/DVT Procedure Form on page~\pageref{Pulmonary embolism/DVT}}
    %\item COPD, emphysema, asthma, or other lung disease %\diagnosisdate
    %\item Infection or sepsis %\diagnosisdate
    \item None of the above
  \vsublistend
\vlistend

\vlist{2.} \item Since the date of your in-person MOSAAIC exam, have you had any of the following procedures? \textit{(Select all that apply)}
\begin{enumerate}[left = 0 em, align = parleft]
    \item Procedure to improve blood flow to the heart (a coronary artery bypass surgery (CABG) or placing a stent in the heart arteries) \ra \textbf{Complete Heart Procedure Form on page~\pageref{Procedure to Increase Blood Flow to Heart}}
    \item Procedure to improve blood flow to the brain \ra \textbf{Complete Brain Procedure Form on page~\pageref{Procedure to Increase Blood Flow to Brain}}
    \item Procedure to improve blood flow to the legs \ra \textbf{Complete Vascular Procedure Form on page~\pageref{Procedure to Increase Blood Flow to Legs}}
    %\item Other surgery (specify:) \hrulefill %\diagnosisdate
    \item None of the above
\vsublistend
\vlistend

If you answered ``None of the above'' to questions 1 and 2, you are finished with Part 1. \textbf{Go to the Health Conditions and Procedures Form (Part 2)}
