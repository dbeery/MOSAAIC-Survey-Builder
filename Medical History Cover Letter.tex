\section*{Cover Letter}

\begin{enumerate}[label = \textbullet] 
    \item Participant Tracking Form: 
        \begin{enumerate}[label = \(\circ\)]
            \item For the electronic version, participant contact information, alternate contacts information into the online survey, and provider information from their Pre-Exam Survey will be pre-loaded, with participants being asked to update/confirm their responses. Therefore, the wording of the prompts and questions will slightly vary.
        \end{enumerate}
    \item Health Conditions and Procedures
        \begin{enumerate}[label = \(\circ\)]
            \item Medical history questions are intended to closely match the medical history questions on the Pre-Exam Survey to minimize translation burden and ensure consistency between surveys. 
            \item For the electronic version, if a participant responsed ``Yes'' to a lifetime diagnosis of a condition in the Pre-Exam Survey, that question will be hidden in this section. 
            \item Participants will be asked if they have been diagnosed with specific conditions before their Pre-Exam Survey date. This date will be written into the paper version by CCFC staff before mailing. For the electronic version, the participant's Pre-Exam Survey date will be pre-loaded.
        \end{enumerate}
    \item Hospitalization Form
        \begin{enumerate}[label = \(\circ\)]
            \item For the electronic version, rather than completing this section and then the Hospitaliation Form, participant's will be prompted to answer questions from the Hospitalization Form if they select ``Yes'' for any question, ``Were you hospitalized for this condition?'', throughout the completion of this section. 
        \end{enumerate}
    \item Death Informant Interview
        \begin{enumerate}[label = \(\circ\)]
            \item Additional information for administering the Death Informant Interview will be included in Volume 2 of the Manual of Procedures. 
        \end{enumerate}  
\end{enumerate}
